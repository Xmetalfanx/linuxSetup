# https://stackoverflow.com/questions/14525296/how-do-i-check-if-variable-is-an-array
function checkIfArray() {
	echo -e "$@"
	sleep 4

	echo -e "Checking if $1 is an array \n "

	if [[ "$(declare -p $1 )" =~ "declare -a" ]]; then
		echo -e "$1 is an array"
	else
		echo -e "$1 is not an array"
	fi

	exit
}
